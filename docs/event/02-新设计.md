# Event 模块新设计

## 1. 设计原则

1. **types 定义数据模型，event 实现实体** — `ncatbot/types/` 是纯 Pydantic 数据模型（可序列化、可独立测试），`ncatbot/event/` 是持有 API 引用的实体类（提供 `reply` 等行为）
2. **`event.reply()` 必须保留** — 插件开发者最重要的 API，通过实体包装实现
3. **消除 Mixin 多继承** — 字段直接在数据模型中声明
4. **OneBot11 wire format 优先** — 消息段中 `type` 字段保留 OB11 `data.type` 原始语义，内部判别用 `_type`
5. **可序列化** — 数据模型可无损 `model_dump()` / `model_validate()`

## 3. types/ — 纯数据模型

### 3.1 BaseEventData

```python
# ncatbot/types/base.py
from pydantic import BaseModel, ConfigDict, model_validator
from .enums import PostType

class BaseEventData(BaseModel):
    """事件数据模型基类 — 纯数据，可序列化"""
    model_config = ConfigDict(extra="allow")

    time: int
    self_id: str
    post_type: PostType

    @model_validator(mode="before")
    @classmethod
    def _coerce_ids(cls, data):
        """统一将所有 *_id 字段从 int/float 转 str"""
        if isinstance(data, dict):
            for key, value in data.items():
                if key.endswith("_id") and isinstance(value, (int, float)):
                    data[key] = str(int(value))
        return data
```

子类无需重复写任何 `field_validator`。

### 3.3 MessageSegment — `_type` 内部判别

**问题回顾**: OB11 消息段格式 `{"type": "music", "data": {"type": "qq", ...}}`，外层 `type` 标识段种类，`data.type` 是段参数。

**方案**: `_type` 是 ClassVar，标识段种类（对应 OB11 外层 type）；`type` 留给 OB11 `data.type`。

```python
# ncatbot/types/segment/base.py
from pydantic import BaseModel, ConfigDict
from typing import ClassVar, Dict, Type

_SEGMENT_MAP: Dict[str, Type["MessageSegment"]] = {}

class MessageSegment(BaseModel):
    """消息段基类"""
    model_config = ConfigDict(extra="allow")
    _type: ClassVar[str]  # 内部判别（OB11 外层 type）

    def __init_subclass__(cls, **kwargs):
        super().__init_subclass__(**kwargs)
        if isinstance(cls.__dict__.get("_type"), str):
            _SEGMENT_MAP[cls._type] = cls

    def to_dict(self) -> dict:
        """序列化为 OB11: {"type": "...", "data": {...}}"""
        return {"type": self._type, "data": self.model_dump(exclude_none=True)}

def parse_segment(raw: dict) -> MessageSegment:
    """从 OB11 {"type": "...", "data": {...}} 解析"""
    seg_type = raw.get("type", "")
    cls = _SEGMENT_MAP.get(seg_type)
    if not cls:
        raise ValueError(f"Unknown segment type: {seg_type}")
    return cls.model_validate(raw.get("data", {}))
```

### 3.5 MessageArray

```python
# ncatbot/types/segment/array.py
class MessageArray:
    """消息段数组"""

    def __init__(self, segments: Optional[List[MessageSegment]] = None):
        self._segments: List[MessageSegment] = list(segments or [])

    @classmethod
    def from_list(cls, data: List[dict]) -> "MessageArray":
        return cls([parse_segment(seg) for seg in data])

    def to_list(self) -> List[dict]:
        return [seg.to_dict() for seg in self._segments]

    # 查询 — T = TypeVar("T", bound=MessageSegment)
    def filter(self, cls: Type[T]) -> List[T]:
        return [s for s in self._segments if isinstance(s, cls)]

    @property
    def text(self) -> str:
        return "".join(s.text for s in self.filter(PlainText))

    def is_at(self, user_id: str, all_except: bool = False) -> bool:
        for at in self.filter(At):
            if at.qq == str(user_id):
                return True
            if at.qq == "all" and not all_except:
                return True
        return False

    # 构造（链式）
    def add_text(self, text: str) -> "MessageArray":
        self._segments.append(PlainText(text=text))
        return self

    def add_image(self, file: str) -> "MessageArray":
        self._segments.append(Image(file=file))
        return self

    def add_at(self, user_id: str) -> "MessageArray":
        self._segments.append(At(qq=str(user_id)))
        return self

    def add_reply(self, message_id: str) -> "MessageArray":
        self._segments.append(Reply(id=str(message_id)))
        return self

    # 容器协议
    def __iter__(self):
        return iter(self._segments)

    def __len__(self):
        return len(self._segments)

    def __add__(self, other):
        return MessageArray(self._segments + list(other))
```

## 4. event/ — 实体层

### 4.1 BaseEvent — 实体基类

```python
# ncatbot/event/base.py
from ncatbot.types import BaseEventData

class BaseEvent:
    """事件实体基类 — 包装数据模型，提供 API 和行为方法"""

    def __init__(self, data: BaseEventData, api: "IBotAPI"):
        self._data = data
        self._api = api

    def __getattr__(self, name: str):
        """代理到数据模型字段，插件代码可直接 event.user_id"""
        try:
            return getattr(self._data, name)
        except AttributeError:
            raise AttributeError(
                f"'{type(self).__name__}' has no attribute '{name}'"
            )

    @property
    def api(self) -> "IBotAPI":
        return self._api

    @property
    def data(self) -> BaseEventData:
        """获取底层数据模型（可序列化）"""
        return self._data
```

### 4.2 MessageEvent

```python
# ncatbot/event/message.py
from ncatbot.types import MessageEventData, GroupMessageEventData

class MessageEvent(BaseEvent):
    """消息事件实体"""
    _data: MessageEventData

    async def reply(self, text, **kwargs):
        if self._data.message_type is MessageType.GROUP:
            return await self._api.send_group_msg(
                group_id=self._data.group_id, message=text, **kwargs
            )
        return await self._api.send_private_msg(
            user_id=self._data.user_id, message=text, **kwargs
        )

    async def delete(self):
        return await self._api.delete_msg(message_id=self._data.message_id)

class PrivateMessageEvent(MessageEvent):
    """私聊消息事件"""
    pass

class GroupMessageEvent(MessageEvent):
    """群消息事件"""
    _data: GroupMessageEventData

    async def kick(self, reject_add_request: bool = False):
        return await self._api.set_group_kick(
            self._data.group_id, self._data.user_id, reject_add_request
        )

    async def ban(self, duration: int = 30 * 60):
        return await self._api.set_group_ban(
            self._data.group_id, self._data.user_id, duration
        )
```

# ncatbot/event/request.py
class RequestEvent(BaseEvent):
    """请求事件实体"""

    async def approve(self, remark: str = "", reason: str = ""):
        if self._data.request_type == "friend":
            return await self._api.set_friend_add_request(
                flag=self._data.flag, approve=True, remark=remark
            )
        elif self._data.request_type == "group":
            return await self._api.set_group_add_request(
                flag=self._data.flag, sub_type=self._data.sub_type,
                approve=True, reason=""

        if self._data.request_type == "friend":
            return await self._api.set_friend_add_request(
                flag=self._data.flag, approve=False
            )
        elif self._data.request_type == "group":
            return await self._api.set_group_add_request(
                flag=self._data.flag, sub_type=self._data.sub_type,
                approve=False, reason=reason

### 4.5 EventParser

```python
# ncatbot/event/parser.py
class EventParser:
    """解析 raw dict → 数据模型 (types/)"""
    _registry: Dict[Tuple[str, str], Type[BaseEventData]] = {}

    @classmethod
    def register(cls, post_type: str, sub_key: str = ""):
        def wrapper(data_cls):
            cls._registry[(post_type, sub_key)] = data_cls
            return data_cls
        return wrapper

    @classmethod
    def parse(cls, data: dict) -> BaseEventData:
        key = cls._resolve_key(data)
        data_cls = cls._registry.get(key)
        if not data_cls:
            raise ValueError(f"Unknown event: {key}")
        return data_cls.model_validate(data)
```

## 5. 数据流

```
Adapter 收到 raw dict
  → EventDispatcher.dispatch(raw)
    → EventParser.parse(raw)              # dict → types/ 数据模型
    → create_entity(data_model, api)      # 数据模型 → event/ 实体
    → BusEvent(type_str, entity)          # 包装为 EventBus 信封
    → EventBus.publish(bus_event)
      → handler(bus_event)                # handler 拿到实体
        → event.reply("hello")           # 直接调用行为方法
        → event.user_id                  # __getattr__ 代理到 data.user_id
        → event.data                     # 获取底层纯数据模型
```

## 7. 与旧版对比

| 对比项 | 旧版 | 新版 |
|--------|------|------|
| `event.reply()` | ✅ Mixin 继承 | ✅ 实体方法 |
| `event.user_id` | 直接字段 | `__getattr__` 代理 |
| `event.data` | 不存在 | 获取纯数据模型 |
| 数据可序列化 | ❌ 持有 `_api` | ✅ `event.data.model_dump()` |
| 独立测试 | ❌ 必须 bind_api | ✅ types/ 无依赖 |
| Mixin 多继承 | ❌ GroupAdminMixin | ✅ 无 Mixin |
| Music/Image type | ❌ 手动重命名 | ✅ `_type` 内部，`type` = OB11 |
| ID validator 重复 | ❌ 每个子类 | ✅ 基类 model_validator |
| 模块位置 | 全部在 event/ | types/ 数据 + event/ 实体 |
